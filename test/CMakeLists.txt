# Tests for POG2WhyML

# Search for a suitable shell program, so that the tests can be run under windows
# if a unix-like shell is available
find_program(SHELL
	NAMES bash sh
	PATHS env PATH)

configure_file(${CMAKE_CURRENT_SOURCE_DIR}/setENV.model
  ${CMAKE_CURRENT_SOURCE_DIR}/setENV)


macro(add_pog2whyml_test id args)
  add_test(POG2WHYML.${id} ${SHELL} ${CMAKE_CURRENT_SOURCE_DIR}/do_ctest ${CMAKE_CURRENT_SOURCE_DIR} ${id})
  set_tests_properties(POG2WHYML.${id} PROPERTIES FAIL_REGULAR_EXPRESSION ".*Error")
  set_tests_properties(POG2WHYML.${id} PROPERTIES PASS_REGULAR_EXPRESSION ".* OK\\.")
  set_tests_properties(POG2WHYML.${id} PROPERTIES TIMEOUT 14400)
  set_tests_properties(POG2WHYML.${id} PROPERTIES LABELS "${args}")
endmacro(add_pog2whyml_test)
